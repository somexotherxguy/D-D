CREATE TABLE users(
	username varchar(30) NOT NULL,
	user_pw varchar(30) NOT NULL,
	player_name varchar(100) NOT NULL,

	PRIMARY KEY (username)
);

CREATE TABLE characters(
	char_name varchar(100) NOT NULL,
	user varchar(30),

	PRIMARY KEY (char_name),
	FOREIGN KEY (user) REFERENCES users(username)

	ON DELETE CASCADE
);

CREATE TABLE basic_info(
	char_name varchar(100) NOT NULL,
	username varchar(30) NOT NULL,
	level int,
	race varchar(30) NOT NULL,
	alignment varchar(30),
	exp int,
	age int,
	height int NOT NULL,
	weight int NOT NULL,
	eyes varchar(30) NOT NULL,
	skin varchar(30) NOT NULL,
	hair varchar(30) NOT NULL,
	hp_total int,
	hp_current int,
	hp_temp int,
	hit_die int,
	armor_class int,
	initiative int,
	speed int,
	Inspiration bool NOT NULL,
	prof int,
	melee_mod int,
	range_mod int,
	
	FOREIGN KEY (char_name) REFERENCES characters(char_name),
	FOREIGN KEY (username) REFERENCES users(username)

	ON DELETE CASCADE
);


CREATE TABLE deathsaves(
	char_name varchar(100) NOT NULL,
	username varchar(30) NOT NULL,
	successes int,
	failures int,

	FOREIGN KEY (char_name) REFERENCES characters(char_name),
	FOREIGN KEY (username) REFERENCES users(username)

	ON DELETE CASCADE
);

CREATE TABLE attributes(
	char_name varchar(100) NOT NULL,
	username varchar(30) NOT NULL,
	str int,
	dex int,
	con int,
	intel int,
	wis int,
	cha int,
	str_mod int,
	dex_mod int,
	con_mod int,
	intel_mod int,
	wis_mod int,
	cha_mod int,

	FOREIGN KEY (char_name) REFERENCES characters(char_name),
	FOREIGN KEY (username) REFERENCES users(username)

	ON DELETE CASCADE
);

CREATE TABLE skills(
	char_name varchar(100) NOT NULL,
	username varchar(30) NOT NULL,
	acro int,
	ani_hand int,
	arcana int,
	athl int,
	decep int,
	hist int,
	insight int,
	intim int,
	invest int,
	medi int,
	nat int,
	percep int,
	perform int,
	pers int,
	reli int,
	soh int,
	stealth int,
	survival int,
	pass_invest int,
	pass_insight int,
	pass_percep int
	acro_prof bool not null,
	ani_hand_prof bool not null,
	arcana_prof bool not null,
	athl_prof bool not null,
	decep_prof bool not null,
	hist_prof bool not null,
	insight_prof bool not null,
	intim_prof bool not null,
	invest_prof bool not null,
	medi_prof bool not null,
	nat_prof bool not null,
	percep_prof bool not null,
	perform_prof bool not null,
	pers_prof bool not null,
	reli_prof bool not null,
	soh_prof bool not null,
	stealth_prof bool not null,
	survival_prof bool not null,
	pass_invest_prof bool not null,
	pass_insight_prof bool not null,
	pass_percep_prof bool not null,

	FOREIGN KEY (char_name) REFERENCES characters(char_name),
	FOREIGN KEY (username) REFERENCES users(username)

);

CREATE TABLE background(
	char_name varchar(100) NOT NULL,
	username varchar(30) NOT NULL,
	personality varchar(1000),
	ideals varchar(1000),
	bonds varchar(1000),
	flaws varchar(1000),
	features_traits varchar(1000),
	feats varchar(1000),
	languages varchar(1000),
	tool_prof varchar(1000),
	backstory varchar(1000),
	weap_prof varchar(1000)

	FOREIGN KEY (char_name) REFERENCES characters(char_name),
	FOREIGN KEY (username) REFERENCES users(username)

	ON DELETE CASCADE
);

CREATE TABLE equipment_treasure(
	char_name varchar(100) NOT NULL,
	username varchar(30) NOT NULL,
	equip varchar(1000),
	treasure varchar(1000),
	plat int,
	gold int,
	silver int,
	copper int,

	FOREIGN KEY (char_name) REFERENCES characters(char_name),
	FOREIGN KEY (username) REFERENCES users(username)

	ON DELETE CASCADE
);

CREATE TABLE saves(
	char_name varchar(100) NOT NULL,
	username varchar(30) NOT NULL,
	str_save int,
	dex_save int,
	con_save int,
	intel_save int,
	wis_save int,
	cha_save int
	str_prof bool NOT NULL,
	dex_prof bool NOT NULL,
	con_prof bool NOT NULL,
	intel_prof bool NOT NULL,
	wis_prof bool NOT NULL,
	cha_prof bool NOT NULL,

	FOREIGN KEY (char_name) REFERENCES characters(char_name),
	FOREIGN KEY (username) REFERENCES users(username)

	ON DELETE CASCADE
);

create table class(
	char_name varchar(100) NOT NULL,
	username varchar(30) NOT NULL,
	rage int,
	inspiration_die int,
	beast_shifts int,
	action_surges int,
	superiority_die int,
	channeled_div int,
	ki_points int,
	lay_on_hands int,
	sorceror_points int,
	invocations int,
	cantrip_slots int,
	one_slot int,
	two_slot int,
	three_slot int,
	four_slot int,
	five_slot int,
	six_slot int,
	seven_slot int,
	eight_slot int,
	nine_slot int,
	spells_known int,
	
	FOREIGN KEY (char_name) REFERENCES characters(char_name),
	FOREIGN KEY (username) REFERENCES users(username)
);

create table feats(
	feat_name varchar(60) NOT NULL,
	prereq_attr varchar(50),
	prereq_caster bool,
	prereq_num int,
	prereq_armor varchar(10),
	description varchar(2000) NOT NULL
);

create table class_features(
	class varchar(30) NOT NULL,
	archetype varchar(30) 
	level int NOT NULL,
	feature_name varchar(30) NOT NULL,
	description varchar(3000) NOT NULL
);
