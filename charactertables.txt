create table characterinfo(
	play_name varchar(30) not null,
	char_name varchar(30) not null,
	level int,
	race varchar(30) not null,
	align varchar(30) not null,
	age int,
	height int not null,
	weight int not null,
	eyes varchar(30) not null,
	skin varchar(30) not null,
	hair varchar(30) not null,
	hp_total int,
	hp_current int,
	temp_hp int,
	hit_die int,
	ac int,
	xp int,
	speed int,
	prof int,
	inspir bool not null,
	init int

);

create table deathsaves(
	play_name varchar(30) not null,
	de_suc int,
	de_fail int
);

create table attributes(
	play_name varchar(30) not null,
	str int,
	dex int,
	con int,
	intel int,
	wis int,
	cha int,
	str_mod int,
	dex_mod int,
	con_mod int,
	intel_mod int,
	wis_mod int,
	cha_mod int

);

	nat_prof bool not null,
	percep_prof bool not null,

create table background_equip(
	play_name varchar(30) not null,
	personality varchar(1000),
	ideals varchar(1000),
	bonds varchar(1000),
	flaws varchar(1000),
	features varchar(1000),
	feats varchar(1000),
	equip varchar(1000),
	plat int,
	gold int,
	silver int,
	copper int,
	languages varchar(1000),
	tool_prof varchar(1000),
	backstory varchar(1000),
	weap_prof varchar(1000)
);

	wis_save int,
	cha_save int,
	str_prof bool not null,
